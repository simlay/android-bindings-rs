26use std::panic::catch_unwind;

28type RunnableClosure = Box<dyn FnMut() + Send + 'static>;
29// Type alias for boxed closure
30
31/// Creates a Java Runnable from a Rust closure
32pub fn create_runnable<'local, F>(
33    env: JNIEnv<'local>,
34    closure: F,
35) -> jni::errors::Result<java::lang::Runnable<'local>>
36where
37    F: FnMut() + Send + 'static,
38{
39    // Box the closure and convert to raw pointer
40    let boxed: RunnableClosure = Box::new(closure);
41    let ptr = Box::into_raw(Box::new(boxed)) as jlong;
42
43    // Find NativeRunnable class and create instance
44    let class = env.find_class("com/example/NativeRunnable")?;
45    let obj = env.new_object(class, "(J)V", &[JValue::Long(ptr)])?;
46
47    Ok(java::lang::Runnable::from(obj))
48}

50/// Native method called when Runnable.run() is invoked
51#[unsafe(no_mangle)]
52pub extern "system" fn Java_com_example_NativeRunnable_nativeRun(
53    _env: JNIEnv,
54    _class: JClass,
55    ptr: jlong,
56) {
57    let _ = catch_unwind(|| {
58        if ptr != 0 {
59            let closure = unsafe { &mut *(ptr as *mut RunnableClosure) };
60            closure();
61        }
62    });
63}

65/// Native method to drop/free the closure when no longer needed
66#[unsafe(no_mangle)]
67pub extern "system" fn Java_com_example_NativeRunnable_nativeDrop(
68    _env: JNIEnv,
69    _class: JClass,
70    ptr: jlong,
71) {
72    if ptr != 0 {
73        unsafe {
74            let _ = Box::from_raw(ptr as *mut RunnableClosure);
75            // Box is dropped here, freeing the closure
76        }
77    }
