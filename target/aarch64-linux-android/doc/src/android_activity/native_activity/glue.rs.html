22#[derive(Clone, Copy, Eq, PartialEq, Debug)]
23pub enum AppCmd {
24    InputQueueChanged = 0,
25    InitWindow = 1,
26    TermWindow = 2,
27    WindowResized = 3,
28    WindowRedrawNeeded = 4,
29    ContentRectChanged = 5,
30    GainedFocus = 6,
31    LostFocus = 7,
32    ConfigChanged = 8,
33    LowMemory = 9,
34    Start = 10,
35    Resume = 11,
36    SaveState = 12,
37    Pause = 13,
38    Stop = 14,
39    Destroy = 15,
40}
41impl TryFrom<i8> for AppCmd {
42    type Error = ();
43
44    fn try_from(value: i8) -> Result<Self, Self::Error> {
45        match value {
46            0 => Ok(AppCmd::InputQueueChanged),
47            1 => Ok(AppCmd::InitWindow),
48            2 => Ok(AppCmd::TermWindow),
49            3 => Ok(AppCmd::WindowResized),
50            4 => Ok(AppCmd::WindowRedrawNeeded),
51            5 => Ok(AppCmd::ContentRectChanged),
52            6 => Ok(AppCmd::GainedFocus),
53            7 => Ok(AppCmd::LostFocus),
54            8 => Ok(AppCmd::ConfigChanged),
55            9 => Ok(AppCmd::LowMemory),
56            10 => Ok(AppCmd::Start),
57            11 => Ok(AppCmd::Resume),
58            12 => Ok(AppCmd::SaveState),
59            13 => Ok(AppCmd::Pause),
60            14 => Ok(AppCmd::Stop),
61            15 => Ok(AppCmd::Destroy),
62            _ => Err(()),
63        }
64    }
65}

99impl NativeActivityGlue {
100    pub fn new(
101        activity: *mut ndk_sys::ANativeActivity,
102        saved_state: *const libc::c_void,
103        saved_state_size: libc::size_t,
104    ) -> Self {
105        let glue = Self {
106            inner: Arc::new(WaitableNativeActivityState::new(
107                activity,
108                saved_state,
109                saved_state_size,
110            )),
111        };
112
113        let weak_ref = Arc::downgrade(&glue.inner);
114        let weak_ptr = Weak::into_raw(weak_ref);
115        unsafe {
116            (*activity).instance = weak_ptr as *mut _;
117
118            (*(*activity).callbacks).onDestroy = Some(on_destroy);
119            (*(*activity).callbacks).onStart = Some(on_start);
120            (*(*activity).callbacks).onResume = Some(on_resume);
121            (*(*activity).callbacks).onSaveInstanceState = Some(on_save_instance_state);
122            (*(*activity).callbacks).onPause = Some(on_pause);
123            (*(*activity).callbacks).onStop = Some(on_stop);
124            (*(*activity).callbacks).onConfigurationChanged = Some(on_configuration_changed);
125            (*(*activity).callbacks).onLowMemory = Some(on_low_memory);
126            (*(*activity).callbacks).onWindowFocusChanged = Some(on_window_focus_changed);
127            (*(*activity).callbacks).onNativeWindowCreated = Some(on_native_window_created);
128            (*(*activity).callbacks).onNativeWindowResized = Some(on_native_window_resized);
129            (*(*activity).callbacks).onNativeWindowRedrawNeeded =
130                Some(on_native_window_redraw_needed);
131            (*(*activity).callbacks).onNativeWindowDestroyed = Some(on_native_window_destroyed);
132            (*(*activity).callbacks).onInputQueueCreated = Some(on_input_queue_created);
133            (*(*activity).callbacks).onInputQueueDestroyed = Some(on_input_queue_destroyed);
134            (*(*activity).callbacks).onContentRectChanged = Some(on_content_rect_changed);
135        }
136
137        glue
138    }

140    /// Returns the file descriptor that needs to be polled by the Rust main thread
141    /// for events/commands from the JVM thread
142    pub fn cmd_read_fd(&self) -> libc::c_int {
143        self.mutex.lock().unwrap().msg_read
144    }

146    /// For the Rust main thread to read a single pending command sent from the JVM main thread
147    pub fn read_cmd(&self) -> Option<AppCmd> {
148        self.inner.mutex.lock().unwrap().read_cmd()
149    }

151    /// For the Rust main thread to get an [`InputQueue`] that wraps the AInputQueue pointer
152    /// we have and at the same time ensure that the input queue is attached to the given looper.
153    ///
154    /// NB: it's expected that the input queue is detached as soon as we know there is new
155    /// input (knowing the app will be notified) and only re-attached when the application
156    /// reads the input (to avoid lots of redundant wake ups)
157    pub fn looper_attached_input_queue(
158        &self,
159        looper: *mut ndk_sys::ALooper,
160        ident: libc::c_int,
161    ) -> Option<InputQueue> {
162        let mut guard = self.mutex.lock().unwrap();
163
164        if guard.input_queue.is_null() {
165            return None;
166        }
167
168        unsafe {
169            // Reattach the input queue to the looper so future input will again deliver an
170            // `InputAvailable` event.
171            guard.attach_input_queue_to_looper(looper, ident);
172            Some(InputQueue::from_ptr(NonNull::new_unchecked(
173                guard.input_queue,
174            )))
175        }
176    }
177
178    pub fn detach_input_queue_from_looper(&self) {
179        unsafe {
180            self.inner
181                .mutex
182                .lock()
183                .unwrap()
184                .detach_input_queue_from_looper();
185        }
186    }
187
188    pub fn config(&self) -> ConfigurationRef {
189        self.mutex.lock().unwrap().config.clone()
190    }
191
192    pub fn content_rect(&self) -> Rect {
193        self.mutex.lock().unwrap().content_rect.into()
194    }

197/// The status of the native thread that's created to run
198/// `android_main`
199#[derive(Copy, Clone, Debug, PartialEq, Eq)]
200pub enum NativeThreadState {
201    /// The `android_main` thread hasn't been created yet
202    Init,
203    /// The `android_main` thread has been spawned and started running
204    Running,
205    /// The `android_main` thread has finished
206    Stopped,
207}

231impl NativeActivityState {
232    pub fn read_cmd(&mut self) -> Option<AppCmd> {
233        let mut cmd_i: i8 = 0;
234        loop {
235            match unsafe { libc::read(self.msg_read, &mut cmd_i as *mut _ as *mut _, 1) } {
236                1 => {
237                    let cmd = AppCmd::try_from(cmd_i);
238                    return match cmd {
239                        Ok(cmd) => Some(cmd),
240                        Err(_) => {
241                            log::error!("Spurious, unknown NativeActivityGlue cmd: {}", cmd_i);
242                            None
243                        }
244                    };
245                }
246                -1 => {
247                    let err = std::io::Error::last_os_error();
248                    if err.kind() != std::io::ErrorKind::Interrupted {
249                        log::error!("Failure reading NativeActivityGlue cmd: {}", err);
250                        return None;
251                    }
252                }
253                count => {
254                    log::error!(
255                        "Spurious read of {count} bytes while reading NativeActivityGlue cmd"
256                    );
257                    return None;
258                }
259            }
260        }
261    }
262
263    fn write_cmd(&mut self, cmd: AppCmd) {
264        let cmd = cmd as i8;
265        loop {
266            match unsafe { libc::write(self.msg_write, &cmd as *const _ as *const _, 1) } {
267                1 => break,
268                -1 => {
269                    let err = std::io::Error::last_os_error();
270                    if err.kind() != std::io::ErrorKind::Interrupted {
271                        log::error!("Failure writing NativeActivityGlue cmd: {}", err);
272                        return;
273                    }
274                }
275                count => {
276                    log::error!(
277                        "Spurious write of {count} bytes while writing NativeActivityGlue cmd"
278                    );
279                    return;
280                }
281            }
282        }
283    }
284
285    pub unsafe fn attach_input_queue_to_looper(
286        &mut self,
287        looper: *mut ndk_sys::ALooper,
288        ident: libc::c_int,
289    ) {
290        if !self.input_queue.is_null() {
291            log::trace!("Attaching input queue to looper");
292            ndk_sys::AInputQueue_attachLooper(
293                self.input_queue,
294                looper,
295                ident,
296                None,
297                ptr::null_mut(),
298            );
299        }
300    }
301
302    pub unsafe fn detach_input_queue_from_looper(&mut self) {
303        if !self.input_queue.is_null() {
304            log::trace!("Detaching input queue from looper");
305            ndk_sys::AInputQueue_detachLooper(self.input_queue);
306        }
307    }
308}

325    pub fn new(
326        activity: *mut ndk_sys::ANativeActivity,
327        saved_state_in: *const libc::c_void,
328        saved_state_size: libc::size_t,
329    ) -> Self {
330        let mut msgpipe: [libc::c_int; 2] = [-1, -1];
331        unsafe {
332            if libc::pipe(msgpipe.as_mut_ptr()) != 0 {
333                panic!(
334                    "could not create  Rust <-> Java IPC pipe: {}",
335                    std::io::Error::last_os_error()
336                );
337            }
338        }
339
340        let saved_state = if saved_state_in.is_null() {
341            Vec::new()
342        } else {
343            unsafe { std::slice::from_raw_parts(saved_state_in as *const u8, saved_state_size) }
344                .to_vec()
345        };
346
347        let config = unsafe {
348            let config = ndk_sys::AConfiguration_new();
349            ndk_sys::AConfiguration_fromAssetManager(config, (*activity).assetManager);
350
351            let config = super::ConfigurationRef::new(Configuration::from_ptr(
352                NonNull::new_unchecked(config),
353            ));
354            log::trace!("Config: {:#?}", config);
355            config
356        };
357
358        Self {
359            activity,
360            mutex: Mutex::new(NativeActivityState {
361                msg_read: msgpipe[0],
362                msg_write: msgpipe[1],
363                config,
364                saved_state,
365                input_queue: ptr::null_mut(),
366                window: None,
367                content_rect: Rect::empty().into(),
368                activity_state: State::Init,
369                destroy_requested: false,
370                thread_state: NativeThreadState::Init,
371                app_has_saved_state: false,
372                destroyed: false,
373                redraw_needed: false,
374                pending_input_queue: ptr::null_mut(),
375                pending_window: None,
376            }),
377            cond: Condvar::new(),
378        }
379    }
380
381    pub fn notify_destroyed(&self) {
382        let mut guard = self.mutex.lock().unwrap();
383        guard.destroyed = true;
384
385        unsafe {
386            guard.write_cmd(AppCmd::Destroy);
387            while guard.thread_state != NativeThreadState::Stopped {
388                guard = self.cond.wait(guard).unwrap();
389            }
390
391            libc::close(guard.msg_read);
392            guard.msg_read = -1;
393            libc::close(guard.msg_write);
394            guard.msg_write = -1;
395        }
396    }

399        let mut guard = self.mutex.lock().unwrap();
400        guard.write_cmd(AppCmd::ConfigChanged);
401    }

408    pub fn notify_focus_changed(&self, focused: bool) {

417    pub fn notify_window_resized(&self, native_window: *mut ndk_sys::ANativeWindow) {
418        let mut guard = self.mutex.lock().unwrap();
419        // set_window always syncs .pending_window back to .window before returning. This callback
420        // from Android can never arrive at an interim state, and validates that Android:
421        // 1. Only provides resizes in between onNativeWindowCreated and onNativeWindowDestroyed;
422        // 2. Doesn't call it on a bogus window pointer that we don't know about.
423        debug_assert_eq!(guard.window.as_ref().unwrap().ptr().as_ptr(), native_window);
424        guard.write_cmd(AppCmd::WindowResized);
425    }
426
427    pub fn notify_window_redraw_needed(&self, native_window: *mut ndk_sys::ANativeWindow) {
428        let mut guard = self.mutex.lock().unwrap();
429        // set_window always syncs .pending_window back to .window before returning. This callback
430        // from Android can never arrive at an interim state, and validates that Android:
431        // 1. Only provides resizes in between onNativeWindowCreated and onNativeWindowDestroyed;
432        // 2. Doesn't call it on a bogus window pointer that we don't know about.
433        debug_assert_eq!(guard.window.as_ref().unwrap().ptr().as_ptr(), native_window);
434        guard.write_cmd(AppCmd::WindowRedrawNeeded);
435    }
436
437    unsafe fn set_input(&self, input_queue: *mut ndk_sys::AInputQueue) {
438        let mut guard = self.mutex.lock().unwrap();
439
440        // The pending_input_queue state should only be set while in this method, and since
441        // it doesn't allow re-entrance and is cleared before returning then we expect
442        // this to be null
443        debug_assert!(
444            guard.pending_input_queue.is_null(),
445            "InputQueue update clash"
446        );
447
448        guard.pending_input_queue = input_queue;
449        guard.write_cmd(AppCmd::InputQueueChanged);
450        while guard.input_queue != guard.pending_input_queue {
451            guard = self.cond.wait(guard).unwrap();
452        }
453        guard.pending_input_queue = ptr::null_mut();
454    }
455
456    unsafe fn set_window(&self, window: Option<NativeWindow>) {
457        let mut guard = self.mutex.lock().unwrap();
458
459        // The pending_window state should only be set while in this method, and since
460        // it doesn't allow re-entrance and is cleared before returning then we expect
461        // this to be None
462        debug_assert!(guard.pending_window.is_none(), "NativeWindow update clash");
463
464        if guard.window.is_some() {
465            guard.write_cmd(AppCmd::TermWindow);
466        }
467        guard.pending_window = window;
468        if guard.pending_window.is_some() {
469            guard.write_cmd(AppCmd::InitWindow);
470        }
471        while guard.window != guard.pending_window {
472            guard = self.cond.wait(guard).unwrap();
473        }
474        guard.pending_window = None;
475    }
476
477    unsafe fn set_content_rect(&self, rect: *const ndk_sys::ARect) {
478        let mut guard = self.mutex.lock().unwrap();
479        guard.content_rect = *rect;
480        guard.write_cmd(AppCmd::ContentRectChanged);
481    }
482
483    unsafe fn set_activity_state(&self, state: State) {
484        let mut guard = self.mutex.lock().unwrap();
485
486        let cmd = match state {
487            State::Init => panic!("Can't explicitly transition into 'init' state"),
488            State::Start => AppCmd::Start,
489            State::Resume => AppCmd::Resume,
490            State::Pause => AppCmd::Pause,
491            State::Stop => AppCmd::Stop,
492        };
493        guard.write_cmd(cmd);
494
495        while guard.activity_state != state {
496            guard = self.cond.wait(guard).unwrap();
497        }
498    }
499
500    fn request_save_state(&self) -> (*mut libc::c_void, libc::size_t) {
501        let mut guard = self.mutex.lock().unwrap();
502
503        // The state_saved flag should only be set while in this method, and since
504        // it doesn't allow re-entrance and is cleared before returning then we expect
505        // this to be None
506        debug_assert!(!guard.app_has_saved_state, "SaveState request clash");
507        guard.write_cmd(AppCmd::SaveState);
508        while !guard.app_has_saved_state {
509            guard = self.cond.wait(guard).unwrap();
510        }
511        guard.app_has_saved_state = false;
512
513        // `ANativeActivity` explicitly documents that it expects save state to be
514        // given via a `malloc()` allocated pointer since it will automatically
515        // `free()` the state after it has been converted to a buffer for the JVM.
516        if !guard.saved_state.is_empty() {
517            let saved_state_size = guard.saved_state.len();
518            let saved_state_src_ptr = guard.saved_state.as_ptr();
519            unsafe {
520                let saved_state = libc::malloc(saved_state_size);
521                assert!(
522                    !saved_state.is_null(),
523                    "Failed to allocate {} bytes for restoring saved application state",
524                    saved_state_size
525                );
526                libc::memcpy(saved_state, saved_state_src_ptr as _, saved_state_size);
527                (saved_state, saved_state_size)
528            }
529        } else {
530            (ptr::null_mut(), 0)
531        }
532    }
533
534    pub fn saved_state(&self) -> Option<Vec<u8>> {
535        let guard = self.mutex.lock().unwrap();
536        if !guard.saved_state.is_empty() {
537            Some(guard.saved_state.clone())
538        } else {
539            None
540        }
541    }
542
543    pub fn set_saved_state(&self, state: &[u8]) {
544        let mut guard = self.mutex.lock().unwrap();
545
546        guard.saved_state.clear();
547        guard.saved_state.extend_from_slice(state);
548    }

566    pub unsafe fn pre_exec_cmd(
567        &self,
568        cmd: AppCmd,
569        looper: *mut ndk_sys::ALooper,
570        input_queue_ident: libc::c_int,
571    ) {
572        log::trace!("Pre: AppCmd::{:#?}", cmd);
573        match cmd {
574            AppCmd::InputQueueChanged => {
575                let mut guard = self.mutex.lock().unwrap();
576                guard.detach_input_queue_from_looper();
577                guard.input_queue = guard.pending_input_queue;
578                if !guard.input_queue.is_null() {
579                    guard.attach_input_queue_to_looper(looper, input_queue_ident);
580                }
581                self.cond.notify_one();
582            }
583            AppCmd::InitWindow => {
584                let mut guard = self.mutex.lock().unwrap();
585                guard.window = guard.pending_window.clone();
586                self.cond.notify_one();
587            }
588            AppCmd::Resume | AppCmd::Start | AppCmd::Pause | AppCmd::Stop => {
589                let mut guard = self.mutex.lock().unwrap();
590                guard.activity_state = match cmd {
591                    AppCmd::Start => State::Start,
592                    AppCmd::Pause => State::Pause,
593                    AppCmd::Resume => State::Resume,
594                    AppCmd::Stop => State::Stop,
595                    _ => unreachable!(),
596                };
597                self.cond.notify_one();
598            }
599            AppCmd::ConfigChanged => {
600                let guard = self.mutex.lock().unwrap();
601                let config = ndk_sys::AConfiguration_new();
602                ndk_sys::AConfiguration_fromAssetManager(config, (*self.activity).assetManager);
603                let config = Configuration::from_ptr(NonNull::new_unchecked(config));
604                guard.config.replace(config);
605                log::debug!("Config: {:#?}", guard.config);
606            }
607            AppCmd::Destroy => {
608                let mut guard = self.mutex.lock().unwrap();
609                guard.destroy_requested = true;
610            }
611            _ => {}
612        }
613    }
614
615    pub unsafe fn post_exec_cmd(&self, cmd: AppCmd) {
616        log::trace!("Post: AppCmd::{:#?}", cmd);
617        match cmd {
618            AppCmd::TermWindow => {
619                let mut guard = self.mutex.lock().unwrap();
620                guard.window = None;
621                self.cond.notify_one();
622            }
623            AppCmd::SaveState => {
624                let mut guard = self.mutex.lock().unwrap();
625                guard.app_has_saved_state = true;
626                self.cond.notify_one();
627            }
628            _ => {}
629        }
630    }

637unsafe fn try_with_waitable_activity_ref(
638    activity: *mut ndk_sys::ANativeActivity,
639    closure: impl FnOnce(Arc<WaitableNativeActivityState>),
640) {
641    assert!(!(*activity).instance.is_null());
642    let weak_ptr: *const WaitableNativeActivityState = (*activity).instance.cast();
643    let weak_ref = Weak::from_raw(weak_ptr);
644    if let Some(waitable_activity) = weak_ref.upgrade() {
645        closure(waitable_activity);
646    } else {
647        log::error!("Ignoring spurious JVM callback after last activity reference was dropped!")
648    }
649    let _ = weak_ref.into_raw();
650}

652unsafe extern "C" fn on_destroy(activity: *mut ndk_sys::ANativeActivity) {
653    abort_on_panic(|| {
654        log::debug!("Destroy: {:p}\n", activity);
655        try_with_waitable_activity_ref(activity, |waitable_activity| {
656            waitable_activity.notify_destroyed()
657        });
658    })
659}

661unsafe extern "C" fn on_start(activity: *mut ndk_sys::ANativeActivity) {
662    abort_on_panic(|| {
663        log::debug!("Start: {:p}\n", activity);
664        try_with_waitable_activity_ref(activity, |waitable_activity| {
665            waitable_activity.set_activity_state(State::Start);
666        });
667    })
668}

670unsafe extern "C" fn on_resume(activity: *mut ndk_sys::ANativeActivity) {
671    abort_on_panic(|| {
672        log::debug!("Resume: {:p}\n", activity);

675        });
676    })

679unsafe extern "C" fn on_save_instance_state(
680    activity: *mut ndk_sys::ANativeActivity,
681    out_len: *mut usize,
682) -> *mut libc::c_void {
683    abort_on_panic(|| {
684        log::debug!("SaveInstanceState: {:p}\n", activity);
685        *out_len = 0;
686        let mut ret = ptr::null_mut();
687        try_with_waitable_activity_ref(activity, |waitable_activity| {
688            let (state, len) = waitable_activity.request_save_state();
689            *out_len = len;
690            ret = state
691        });
692
693        log::debug!("Saved state = {:p}, len = {}", ret, *out_len);
694        ret
695    })
696}

827/// This is the native entrypoint for our cdylib library that `ANativeActivity` will look for via `dlsym`
828#[no_mangle]
829extern "C" fn ANativeActivity_onCreate(
830    activity: *mut ndk_sys::ANativeActivity,
831    saved_state: *const libc::c_void,
832    saved_state_size: libc::size_t,
833) {
834    abort_on_panic(|| {
835        let _join_log_forwarder = forward_stdio_to_logcat();
836
837        log::trace!(
838            "Creating: {:p}, saved_state = {:p}, save_state_size = {}",
839            activity,
840            saved_state,
841            saved_state_size
842        );
843
844        // Conceptually we associate a glue reference with the JVM main thread, and another
845        // reference with the Rust main thread
846        let jvm_glue = NativeActivityGlue::new(activity, saved_state, saved_state_size);
847
848        let rust_glue = jvm_glue.clone();
849        // Let us Send the NativeActivity pointer to the Rust main() thread without a wrapper type
850        let activity_ptr: libc::intptr_t = activity as _;
851
852        // Note: we drop the thread handle which will detach the thread
853        std::thread::spawn(move || {
854            let activity: *mut ndk_sys::ANativeActivity = activity_ptr as *mut _;
855
856            let jvm = abort_on_panic(|| unsafe {
857                let na = activity;
858                let jvm: *mut jni_sys::JavaVM = (*na).vm;
859                let activity = (*na).clazz; // Completely bogus name; this is the _instance_ not class pointer
860                ndk_context::initialize_android_context(jvm.cast(), activity.cast());
861
862                let jvm = CloneJavaVM::from_raw(jvm).unwrap();
863                // Since this is a newly spawned thread then the JVM hasn't been attached
864                // to the thread yet. Attach before calling the applications main function
865                // so they can safely make JNI calls
866                jvm.attach_current_thread_permanently().unwrap();
867                jvm
868            });
869
870            let app = AndroidApp::new(rust_glue.clone(), jvm.clone());
871
872            rust_glue.notify_main_thread_running();
873
874            unsafe {
875                // Name thread - this needs to happen here after attaching to a JVM thread,
876                // since that changes the thread name to something like "Thread-2".
877                let thread_name = std::ffi::CStr::from_bytes_with_nul(b"android_main\0").unwrap();
878                libc::pthread_setname_np(libc::pthread_self(), thread_name.as_ptr());
879
880                // We want to specifically catch any panic from the application's android_main
881                // so we can finish + destroy the Activity gracefully via the JVM
882                catch_unwind(|| {
883                    // XXX: If we were in control of the Java Activity subclass then
884                    // we could potentially run the android_main function via a Java native method
885                    // springboard (e.g. call an Activity subclass method that calls a jni native
886                    // method that then just calls android_main()) that would make sure there was
887                    // a Java frame at the base of our call stack which would then be recognised
888                    // when calling FindClass to lookup a suitable classLoader, instead of
889                    // defaulting to the system loader. Without this then it's difficult for native
890                    // code to look up non-standard Java classes.
891                    android_main(app);
892                })
893                .unwrap_or_else(log_panic);
894
895                // Let JVM know that our Activity can be destroyed before detaching from the JVM
896                //
897                // "Note that this method can be called from any thread; it will send a message
898                //  to the main thread of the process where the Java finish call will take place"
899                ndk_sys::ANativeActivity_finish(activity);
900
901                // This should detach automatically but lets detach explicitly to avoid depending
902                // on the TLS trickery in `jni-rs`
903                jvm.detach_current_thread();
904
905                ndk_context::release_android_context();
906            }
907
908            rust_glue.notify_main_thread_stopped_running();
909        });
910
911        // Wait for thread to start.
912        let mut guard = jvm_glue.mutex.lock().unwrap();
913
914        // Don't specifically wait for `Running` just in case `android_main` returns
915        // immediately and the state is set to `Stopped`
916        while guard.thread_state == NativeThreadState::Init {
917            guard = jvm_glue.cond.wait(guard).unwrap();
918        }
919    })
920}
