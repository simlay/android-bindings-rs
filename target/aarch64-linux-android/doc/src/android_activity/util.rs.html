12pub fn try_get_path_from_ptr(path: *const c_char) -> Option<std::path::PathBuf> {
13    if path.is_null() {
14        return None;
15    }
16    let cstr = unsafe {
17        let cstr_slice = CStr::from_ptr(path.cast());
18        cstr_slice.to_str().ok()?
19    };
20    if cstr.is_empty() {
21        return None;
22    }
23    Some(std::path::PathBuf::from(cstr))
24}

26pub(crate) fn android_log(level: Level, tag: &CStr, msg: &CStr) {
27    let prio = match level {
28        Level::Error => ndk_sys::android_LogPriority::ANDROID_LOG_ERROR,
29        Level::Warn => ndk_sys::android_LogPriority::ANDROID_LOG_WARN,
30        Level::Info => ndk_sys::android_LogPriority::ANDROID_LOG_INFO,
31        Level::Debug => ndk_sys::android_LogPriority::ANDROID_LOG_DEBUG,
32        Level::Trace => ndk_sys::android_LogPriority::ANDROID_LOG_VERBOSE,
33    };
34    unsafe {
35        ndk_sys::__android_log_write(prio.0 as libc::c_int, tag.as_ptr(), msg.as_ptr());
36    }
37}

39pub(crate) fn forward_stdio_to_logcat() -> std::thread::JoinHandle<Result<()>> {
40    // XXX: make this stdout/stderr redirection an optional / opt-in feature?...
41
42    let file = unsafe {
43        let mut logpipe: [RawFd; 2] = Default::default();
44        libc::pipe2(logpipe.as_mut_ptr(), libc::O_CLOEXEC);
45        libc::dup2(logpipe[1], libc::STDOUT_FILENO);
46        libc::dup2(logpipe[1], libc::STDERR_FILENO);
47        libc::close(logpipe[1]);
48
49        File::from_raw_fd(logpipe[0])
50    };
51
52    std::thread::Builder::new()
53        .name("stdio-to-logcat".to_string())
54        .spawn(move || -> Result<()> {
55            let tag = CStr::from_bytes_with_nul(b"RustStdoutStderr\0").unwrap();
56            let mut reader = BufReader::new(file);
57            let mut buffer = String::new();
58            loop {
59                buffer.clear();
60                let len = match reader.read_line(&mut buffer) {
61                    Ok(len) => len,
62                    Err(e) => {
63                        error!("Logcat forwarder failed to read stdin/stderr: {e:?}");
64                        break Err(e);
65                    }
66                };
67                if len == 0 {
68                    break Ok(());
69                } else if let Ok(msg) = CString::new(buffer.clone()) {
70                    android_log(Level::Info, tag, &msg);
71                }
72            }
73        })
74        .expect("Failed to start stdout/stderr to logcat forwarder thread")
75}

77pub(crate) fn log_panic(panic: Box<dyn std::any::Any + Send>) {
78    let rust_panic = unsafe { CStr::from_bytes_with_nul_unchecked(b"RustPanic\0") };
79
80    if let Some(panic) = panic.downcast_ref::<String>() {
81        if let Ok(msg) = CString::new(panic.clone()) {
82            android_log(Level::Error, rust_panic, &msg);
83        }
84    } else if let Ok(panic) = panic.downcast::<&str>() {
85        if let Ok(msg) = CString::new(*panic) {
86            android_log(Level::Error, rust_panic, &msg);
87        }
88    } else {
89        let unknown_panic = unsafe { CStr::from_bytes_with_nul_unchecked(b"UnknownPanic\0") };
90        android_log(Level::Error, unknown_panic, unsafe {
91            CStr::from_bytes_with_nul_unchecked(b"\0")
92        });
93    }
94}

96/// Run a closure and abort the program if it panics.
97///
98/// This is generally used to ensure Rust callbacks won't unwind past the JNI boundary, which leads
99/// to undefined behaviour.
100///
101/// TODO(rib): throw a Java exception instead of aborting. An Android Activity does not necessarily
102/// own the entire process because other application Services (or even Activities) may run in
103/// threads within the same process, and so we're tearing down too much by aborting the process.
104pub(crate) fn abort_on_panic<R>(f: impl FnOnce() -> R) -> R {
105    std::panic::catch_unwind(std::panic::AssertUnwindSafe(f)).unwrap_or_else(|panic| {
106        // Try logging the panic before aborting
107        //
108        // Just in case our attempt to log a panic could itself cause a panic we use a
109        // second catch_unwind here.
110        let _ = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| log_panic(panic)));
111        std::process::abort();
112    })
113}
