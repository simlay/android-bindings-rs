6/// Lifetime'd representation of a `jstring`. Just a `JObject` wrapped in a new

56impl<'local> JString<'local> {
57    /// Creates a [`JString`] that wraps the given `raw` [`jstring`]
58    ///
59    /// # Safety
60    ///
61    /// `raw` may be a null pointer. If `raw` is not a null pointer, then:
62    ///
63    /// * `raw` must be a valid raw JNI local reference.
64    /// * There must not be any other `JObject` representing the same local reference.
65    /// * The lifetime `'local` must not outlive the local reference frame that the local reference
66    ///   was created in.
67    pub unsafe fn from_raw(raw: jstring) -> Self {
68        Self(JObject::from_raw(raw as jobject))
69    }
70
71    /// Unwrap to the raw jni type.
72    pub fn into_raw(self) -> jstring {
73        self.0.into_raw() as jstring
74    }
