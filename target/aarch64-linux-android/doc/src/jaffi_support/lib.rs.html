22pub(crate) fn get_class_name<'j>(
23    env: JNIEnv<'j>,
24    clazz: JClass<'j>,
25) -> Result<String, jni::errors::Error> {
26    let name = env.call_method(clazz, "getCanonicalName", "()Ljava/lang/String;", &[])?;
27    let name = name.l()?;
28    let name = JString::from(name);
29    let name = env.get_string(name)?;
30    Ok(Cow::from(&name).to_string())
31}
32
33fn call_string_method<'j, 'l: 'j>(
34    env: &'l JNIEnv<'j>,
35    obj: JObject<'j>,
36    method: &str,
37) -> Result<Option<JavaStr<'j, 'l>>, jni::errors::Error> {
38    let jstring = env
39        .call_method(obj, method, "()Ljava/lang/String;", &[])?
40        .l()
41        .map(JString::from)?;
42
43    if jstring.is_null() {
44        return Ok(None);
45    }
46
47    env.get_string(jstring).map(Some)
48}

231/// Strings
232impl<'j, J> FromJavaToRust<'j, J> for String
233where
234    J: 'j + Deref<Target = JObject<'j>>,
235{
236    // TODO: there's probably a somewhat cheaper option to reduce all the allocations here.
237    fn java_to_rust(java: J, env: JNIEnv<'j>) -> Self {
238        // We're going to have Java properly return utf-8 bytes from a String rather than the BS that is the "reduced utf-8" in JNI
239        let utf8_arg = env
240            .new_string("UTF-8")
241            .expect("Java couldn't allocate a simple string");
242
243        // TODO: cache the method_id...
244        let byte_array = env
245            .call_method(
246                *java,
247                "getBytes",
248                "(Ljava/lang/String;)[B",
249                &[JValue::Object(utf8_arg.into())],
250            )
251            .expect("couldn't call a standard method in Java");
252        let byte_array = byte_array
253            .l()
254            .expect("should have been a JObject of a byte array");
255
256        let bytes = env
257            .convert_byte_array(*byte_array)
258            .expect("the byte_array from previous call was bad");
259
260        // Java should really not have returned bad UTF-8
261        unsafe { String::from_utf8_unchecked(bytes) }
262    }
263}

273where
274    S: KnownString,
275{
276    fn rust_to_java(rust: S, env: JNIEnv<'j>) -> Self {

282/// Convert from a JValue (return type in Java) into the Rust type
283///
284/// This is infallible because the generated code using it should "know" that the type is already correct
285pub trait FromJavaValue<'j, J>: Sized {
286    fn from_jvalue(env: JNIEnv<'j>, jvalue: JValue<'j>) -> Self;
287}
288
289impl<'j, T, J> FromJavaValue<'j, J> for T
290where
291    T: FromJavaToRust<'j, J>,
292    J: 'j,
293    J: From<JObject<'j>>,
294{
295    fn from_jvalue(env: JNIEnv<'j>, jvalue: JValue<'j>) -> Self {
296        let object = jvalue.l().expect("wrong type conversion");
297        Self::java_to_rust(object.into(), env)
298    }
299}

301macro_rules! from_java_value {
302    ($jtype: ident, $rtype:ty, $jval_func: ident) => {
303        impl<'j> FromJavaValue<'j, $jtype> for $rtype {
304            fn from_jvalue(env: JNIEnv<'j>, jvalue: JValue<'j>) -> Self {
305                let t = $jtype(jvalue.$jval_func().expect("wrong type conversion"));
306                Self::java_to_rust(t, env)
307            }
308        }
309    };
310}
311
312impl<'j> FromJavaValue<'j, JavaBoolean> for bool {
313    fn from_jvalue(_env: JNIEnv<'j>, jvalue: JValue<'j>) -> Self {
314        jvalue.z().expect("wrong type conversion")
315    }
316}

356macro_rules! into_java_value {
357    ($jtype: ident, $rtype:ty) => {
358        impl IntoJavaValue<'_, $jtype> for $rtype {
359            fn into_java_value(self, env: JNIEnv<'_>) -> JValue<'_> {
